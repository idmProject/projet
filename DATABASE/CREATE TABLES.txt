CREATE TABLE Ville
(
	VilleId INT IDENTITY NOT NULL UNIQUE,
	Ville NVARCHAR(50) NOT NULL,
	CONSTRAINT VilleId_PK PRIMARY KEY (VilleId)
)
GO
CREATE TABLE Site
(
	SiteId INT IDENTITY NOT NULL UNIQUE,
	Adresse NVARCHAR(50) NULL,
	Alias NVARCHAR(50) NULL,
	VilleId INT NOT NULL,
	CONSTRAINT SiteId_PK PRIMARY KEY(SiteId),
	CONSTRAINT Site_Vill_FK FOREIGN KEY(VilleId)
	REFERENCES Ville(VilleId)
)
GO
CREATE TABLE TypeEspace
(
	TypeEspaceId INT NOT NULL IDENTITY UNIQUE,
	Description NVARCHAR(50) NULL,
	Prix MONEY NULL,
	SiteId INT NOT NULL,
	Largeur INT NULL,
	Longueur INT NULL,
	CONSTRAINT TypeEspaceId_PK PRIMARY KEY(TypeEspaceId),
	CONSTRAINT TypeEspace_Site_FK FOREIGN KEY(SiteId)
	REFERENCES  Site(SiteId) ON DELETE CASCADE
)
GO
CREATE TABLE Espace 
(
	EspaceId INT IDENTITY NOT NULL UNIQUE,
	TypeEspaceId INT NOT NULL,
	Disponible BIT NOT NULL,
	CONSTRAINT EspaceId_PK PRIMARY KEY (EspaceId),
	CONSTRAINT Espace_TypeEspace_FK FOREIGN KEY (TypeEspaceId)
	REFERENCES TypeEspace(TypeEspaceId) ON DELETE CASCADE
)
GO
CREATE TABLE MethodePaiement
(
	MethodePaiementId INT IDENTITY NOT NULL UNIQUE,
	MethodeDescription NVARCHAR(50) NOT NULL,
	
	CONSTRAINT MethodePaiement_PK PRIMARY KEY (MethodePaiementId)
)
GO
CREATE TABLE Client
(
	ClientId INT IDENTITY NOT NULL UNIQUE,
	Actif BIT NOT NULL,
	Adresse NVARCHAR(50) NULL,
	VilleId INT NULL,
	DateEntree DATETIME NULL,
	Nom NVARCHAR(50) NOT NULL,
	Commentaire NVARCHAR(500) NULL,
	MethodePaiementId INT NULL,
	Prenom NVARCHAR(50) NULL,
	Solde MONEY NULL,
	CONSTRAINT ClientId_PK PRIMARY KEY(ClientId),
	CONSTRAINT Client_Ville_FK FOREIGN KEY (VilleId)
	REFERENCES Ville(VilleId),
	CONSTRAINT Client_Methode_FK FOREIGN KEY (MethodePaiementId)
	REFERENCES MethodePaiement(MethodePaiementId)
)
GO
CREATE TABLE Telephone
(
	TelephoneId INT IDENTITY NOT NULL UNIQUE,
	NoTel NVARCHAR(50) NOT NULL,
	Contact NVARCHAR(50) NOT NULL,
	Description NVARCHAR(50) NULL,
	ClientId INT NOT NULL,
	CONSTRAINT TelephoneId_PK PRIMARY KEY(TelephoneId)
)
GO
CREATE TABLE Taxe
(
	--TaxeId INT IDENTITY NOT NULL UNIQUE,
	DateModification DATE NOT NULL,
	Taux FLOAT(53) NOT NULL,
	Type NVARCHAR(50) NOT NULL,
	--CONSTRAINT TaxeId_PK PRIMARY KEY(TaxeId)
	CONSTRAINT DateModification_PK PRIMARY KEY(DateModification, Type)
)
GO
CREATE TABLE Paiement
(
	PaiementId INT IDENTITY NOT NULL UNIQUE,
	DatePaiement DATETIME NOT NULL,
	MontantPaye MONEY NOT NULL,
	ClientId INT NOT NULL,
	Commentaire NVARCHAR(500) NULL,
	MethodePaiementId INT NOT NULL,
	--CONSTRAINT Paiement_MethdoePaiement_FK FOREIGN KEY (MethodePaiementId)
	--REFERENCES MethodePaiement(MethodePaiementId),
	CONSTRAINT PaiementId_PK PRIMARY KEY(PaiementId),
	CONSTRAINT Paiement_Client_FK FOREIGN KEY (ClientId)
	REFERENCES Client(ClientId)
)

GO

CREATE TABLE CodeElectronique
(
	CodeId INT IDENTITY NOT NULL UNIQUE,
	NoCode INT NOT NULL,
	TypeCode NVARCHAR(50) NULL,
	ClientId INT NULL,
	SiteId INT NOT NULL,
	CONSTRAINT CodeId_PK PRIMARY KEY(CodeId),
	CONSTRAINT CodeElectronique_Client_FK FOREIGN KEY (ClientId)
	REFERENCES Client(ClientId) ON DELETE CASCADE,
	CONSTRAINT CodeElectronique_Site_FK FOREIGN KEY (SiteId)
	REFERENCES Site(SiteId)
)

GO

CREATE TABLE Contrat
(
	ContratId INT IDENTITY NOT NULL UNIQUE,
	EspaceId INT NOT NULL,
	DateEntree DATETIME NOT NULL,
	DateSortie DATETIME NULL,
	Prix MONEY NOT NULL,
	ClientId INT NOT NULL,
	CONSTRAINT ContratId_PK PRIMARY KEY (ContratId),
	CONSTRAINT Contrat_Client_FK FOREIGN KEY (ClientId)
	REFERENCES Client(ClientId),
	CONSTRAINT Contrat_Espace_FK FOREIGN KEY (EspaceId)
	REFERENCES Espace(EspaceId)
)

GO

CREATE TABLE Stationnement
(
	EspaceId INT NOT NULL,

CREATE VIEW DossierStationnementView AS
SELECT S.EspaceId, S.NoStationnement, E.Disponible, T.Description, T.Largeur, T.Longueur, T.Prix, Si.Alias, V.Ville FROM Stationnement S
JOIN Espace E ON S.EspaceId = E.EspaceId
JOIN TypeEspace T ON E.TypeEspaceId = T.TypeEspaceId
JOIN Site Si ON  T.SiteId = Si.SiteId
JOIN Ville V ON V.VilleId = Si.VilleId

CREATE VIEW DossiersLocalView AS
SELECT L.NoLocal, E.Disponible, T.Description, T.Largeur, T.Longueur, T.Prix, S.Alias, V.Ville FROM Local L
JOIN Espace E ON L.EspaceId = E.EspaceId
JOIN TypeEspace T ON E.TypeEspaceId = T.TypeEspaceId
JOIN Site S ON  T.SiteId = S.SiteId
JOIN Ville V ON V.VilleId = S.VilleId
